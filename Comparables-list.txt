
public class attributeoptionlist
    {
        public string attributeids { get; set; }
        public int attributeID { get; set; }
        public string attributename { get; set; }
        public string isdefault { get; set; }

    }
 
    public class ViewOption:IViewOption
    {
        #region private variables

        private SVModel.SVDataContainer _dataContext = null;
      

        #endregion

        #region Methods

        public List<attributeoptionlist> SetViewOptionAttributes(Int64 optionID)
        {
            List<attributeoptionlist> objoptionAttributes=new List<attributeoptionlist> ();
 
            _dataContext = new SVModel.SVDataContainer();

            var optionAttribute =     from viewoptionattributes in _dataContext.ViewOptionAttributes
                                      join userviewoptionattributes in _dataContext.UserViewOptionAttributes
                                      on viewoptionattributes.OptionID equals userviewoptionattributes.OptionID into resultTable
                                      where viewoptionattributes.OptionID == optionID || viewoptionattributes.OptionID == null
                                      from userviewoptionattributes in resultTable.DefaultIfEmpty()
                                      select new attributeoptionlist
                                      {
                                          attributeids = userviewoptionattributes != null ? userviewoptionattributes.AttributeIDs : null,
                                          attributeID=viewoptionattributes.AttributeID,
                                          attributename=viewoptionattributes.AttributeName,
                                          isdefault = viewoptionattributes!=null ? viewoptionattributes.IsDefault : null
                                      };
            objoptionAttributes=optionAttribute.ToList();

            return objoptionAttributes;
        }
